## 🐍 代码模板

### **Python自动化脚本模板**：
```python
#!/usr/bin/env python3
"""
AI内容生成与分发系统 - 自动化脚本
版本：1.0
功能：自动生成内容并发布到多个平台
"""

import os
import json
import requests
from datetime import datetime
import logging

# 配置日志
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger(__name__)

class ContentAutomationSystem:
    def __init__(self, config_file="config.json"):
        """初始化自动化系统"""
        self.config = self.load_config(config_file)
        self.today = datetime.now().strftime("%Y-%m-%d")
        
    def load_config(self, config_file):
        """加载配置文件"""
        try:
            with open(config_file, 'r') as f:
                return json.load(f)
        except FileNotFoundError:
            logger.error(f"配置文件 {config_file} 不存在")
            return {
                "openai_api_key": os.getenv("OPENAI_API_KEY", ""),
                "wordpress_url": os.getenv("WORDPRESS_URL", ""),
                "wordpress_username": os.getenv("WORDPRESS_USERNAME", ""),
                "wordpress_password": os.getenv("WORDPRESS_PASSWORD", ""),
                "output_dir": "generated_content"
            }
    
    def generate_content(self, topic, word_count=1000):
        """使用AI生成内容"""
        logger.info(f"开始生成主题为'{topic}'的内容")
        
        # 这里应该调用AI API，简化示例
        content = f"""
        # {topic}
        
        这是关于{topic}的自动生成内容，生成时间：{self.today}。
        
        ## 主要内容
        
        这是一个示例内容段落，实际应用中会调用GPT-4或Claude-3 API来生成高质量内容。
        
        ## 关键要点
        
        1. 自动化内容生成可以节省大量时间
        2. AI生成的内容质量越来越高
        3. 多平台发布可以最大化覆盖面
        
        ## 结论
        
        通过自动化系统，您可以24/7生成和发布内容，建立稳定的被动收入流。
        """
        
        # 保存内容到文件
        output_file = f"{self.config['output_dir']}/{self.today}_{topic.replace(' ', '_')}.md"
        os.makedirs(os.path.dirname(output_file), exist_ok=True)
        
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(content)
        
        logger.info(f"内容已保存到: {output_file}")
        return output_file
    
    def publish_to_wordpress(self, content_file, title, category="AI内容"):
        """发布内容到WordPress"""
        logger.info(f"发布内容到WordPress: {title}")
        
        # WordPress REST API发布
        wp_url = f"{self.config['wordpress_url']}/wp-json/wp/v2/posts"
        
        with open(content_file, 'r', encoding='utf-8') as f:
            content = f.read()
        
        post_data = {
            "title": title,
            "content": content,
            "status": "publish",
            "categories": [category]
        }
        
        try:
            response = requests.post(
                wp_url,
                json=post_data,
                auth=(self.config['wordpress_username'], self.config['wordpress_password'])
            )
            
            if response.status_code == 201:
                logger.info(f"成功发布到WordPress: {response.json().get('link')}")
                return True
            else:
                logger.error(f"WordPress发布失败: {response.status_code} - {response.text}")
                return False
                
        except Exception as e:
            logger.error(f"发布过程中出错: {str(e)}")
            return False
    
    def run_daily_automation(self):
        """运行每日自动化任务"""
        logger.info("开始每日自动化任务")
        
        # 1. 生成今日内容
        topics = ["AI自动化赚钱", "数字营销策略", "被动收入系统"]
        
        for topic in topics:
            content_file = self.generate_content(topic)
            
            # 2. 发布到WordPress
            title = f"{topic} - {self.today}"
            self.publish_to_wordpress(content_file, title)
            
            # 3. 记录到日志
            self.log_activity(topic, content_file)
        
        logger.info("每日自动化任务完成")
    
    def log_activity(self, topic, content_file):
        """记录活动到日志文件"""
        log_entry = {
            "date": self.today,
            "topic": topic,
            "content_file": content_file,
            "timestamp": datetime.now().isoformat()
        }
        
        log_file = "automation_log.json"
        logs = []
        
        if os.path.exists(log_file):
            with open(log_file, 'r') as f:
                logs = json.load(f)
        
        logs.append(log_entry)
        
        with open(log_file, 'w') as f:
            json.dump(logs, f, indent=2, ensure_ascii=False)
        
        logger.info(f"活动已记录: {topic}")

# 使用示例
if __name__ == "__main__":
    # 创建自动化系统实例
    automation = ContentAutomationSystem()
    
    # 运行每日任务
    automation.run_daily_automation()
    
    print("自动化系统运行完成！")
```
