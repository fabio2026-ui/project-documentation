    def __init__(self, name, income_source):
        self.name = name
        self.income_source = income_source
        self.is_running = False
    
    def start(self):
        self.is_running = True
        print(f"启动管道: {self.name}")
    
    def stop(self):
        self.is_running = False
        print(f"停止管道: {self.name}")
    
    def get_revenue(self):
        # 计算当前收入
        return self.income_source.calculate()

### 2. 监控系统实现
```python
# 管道监控系统
class PipelineMonitor:
    def __init__(self):
        self.metrics = {}
    
    def track_metric(self, pipeline_name, metric, value):
        if pipeline_name not in self.metrics:
            self.metrics[pipeline_name] = {}
        self.metrics[pipeline_name][metric] = value
    
    def get_status(self, pipelines):
        status = {}
        for pipeline in pipelines:
            status[pipeline.name] = {
                "running": pipeline.is_running,
                "revenue": pipeline.get_revenue() if hasattr(pipeline, 'get_revenue') else 0,
                "health": self.check_health(pipeline)
            }
        return status
    
    def check_health(self, pipeline):
        # 检查管道健康状态
        return "healthy" if pipeline.is_running else "stopped"
```
