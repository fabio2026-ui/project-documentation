## 🐍 自动化评估工具

### **Python评估脚本** (`project_evaluator.py`)：
```python
#!/usr/bin/env python3
"""
赚钱项目自动化评估工具
功能：自动评估项目的优先级和可行性
"""

import json
from datetime import datetime

class ProjectEvaluator:
    def __init__(self):
        self.projects = []
        self.evaluation_criteria = {
            "market_size": {"weight": 0.3, "min": 1, "max": 10},
            "profit_margin": {"weight": 0.25, "min": 1, "max": 10},
            "startup_speed": {"weight": 0.2, "min": 1, "max": 10},
            "tech_feasibility": {"weight": 0.15, "min": 1, "max": 10},
            "competitive_edge": {"weight": 0.1, "min": 1, "max": 10}
        }
    
    def add_project(self, name, scores):
        """添加项目并计算优先级分数"""
        if not all(key in scores for key in self.evaluation_criteria):
            raise ValueError("评分必须包含所有评估标准")
        
        # 计算加权分数
        priority_score = 0
        for criterion, config in self.evaluation_criteria.items():
            score = scores.get(criterion, 1)
            # 确保分数在范围内
            score = max(config["min"], min(config["max"], score))
            priority_score += score * config["weight"]
        
        project = {
            "name": name,
            "scores": scores,
            "priority_score": round(priority_score, 2),
            "added_date": datetime.now().isoformat()
        }
        
        self.projects.append(project)
        return project
    
    def get_ranked_projects(self):
        """获取按优先级排序的项目列表"""
        return sorted(self.projects, key=lambda x: x["priority_score"], reverse=True)
    
    def generate_report(self):
        """生成评估报告"""
        ranked = self.get_ranked_projects()
        
        report = {
            "generated_date": datetime.now().isoformat(),
            "total_projects": len(self.projects),
            "evaluation_criteria": self.evaluation_criteria,
            "ranked_projects": ranked,
            "recommendations": []
        }
        
        # 生成推荐
        if ranked:
            top_project = ranked[0]
            report["recommendations"].append({
                "action": "立即启动",
                "project": top_project["name"],
                "reason": f"优先级分数最高: {top_project['priority_score']}"
            })
            
            if len(ranked) >= 2:
                second_project = ranked[1]
                report["recommendations"].append({
                    "action": "并行准备",
                    "project": second_project["name"],
                    "reason": f"优先级分数第二: {second_project['priority_score']}"
                })
        
        return report
    
    def save_report(self, filename="project_evaluation_report.json"):
        """保存评估报告到文件"""
        report = self.generate_report()
        
        with open(filename, 'w', encoding='utf-8') as f:
            json.dump(report, f, indent=2, ensure_ascii=False)
        
        print(f"评估报告已保存到: {filename}")
        return filename

# 使用示例
if __name__ == "__main__":
    # 创建评估器
    evaluator = ProjectEvaluator()
    
    # 添加示例项目
    evaluator.add_project("AI内容生成系统", {
        "market_size": 9,
        "profit_margin": 8,
        "startup_speed": 9,
        "tech_feasibility": 8,
        "competitive_edge": 7
    })
    
    evaluator.add_project("AI交易信号系统", {
        "market_size": 10,
        "profit_margin": 9,
        "startup_speed": 7,
        "tech_feasibility": 6,
        "competitive_edge": 8
    })
    
    evaluator.add_project("AI客户服务系统", {
        "market_size": 8,
        "profit_margin": 7,
        "startup_speed": 8,
        "tech_feasibility": 8,
        "competitive_edge": 6
    })
    
    # 获取排名
    ranked = evaluator.get_ranked_projects()
    print("项目优先级排名:")
    for i, project in enumerate(ranked, 1):
        print(f"{i}. {project['name']}: {project['priority_score']}")
    
    # 生成并保存报告
    evaluator.save_report()
```

def evaluate_project(project_data):
    score = 0
    # 评估逻辑
    score += project_data.get("automation_potential", 0) * 0.3
    score += project_data.get("revenue_potential", 0) * 0.25
    score += project_data.get("implementation_speed", 0) * 0.2
    score += project_data.get("competitive_advantage", 0) * 0.15
    score += project_data.get("risk_level", 0) * 0.1
    
    if score >= 0.8:
        return "强烈推荐"
    elif score >= 0.6:
        return "推荐"
    else:
        return "谨慎考虑"
